```python
async def check_approval(self, context: PipelineContext):
    """
    Verifica se requer aprovação manual E commita código no repositório.
    """
    # ... código existente de quality_score ...

    if quality_score >= self.auto_approval_threshold and not has_critical:
        logger.info('auto_approved', score=quality_score)
        context.metadata['approval'] = 'auto'
        # Commitar código mesmo quando auto-approved
        await self._commit_and_push(context, auto_approved=True)

    elif quality_score < self.min_quality_score or has_critical:
        logger.warning('auto_rejected', score=quality_score, critical=has_critical)
        context.metadata['approval'] = 'rejected'

    else:
        # Requer revisão manual - commitar e criar MR
        logger.info('requires_manual_review', score=quality_score)

        # NOVO: Commit e push antes de criar MR
        branch_name = await self._commit_and_push(context, auto_approved=False)

        # Criar MR com branch já contendo código
        mr_url = await self.git_client.create_merge_request(
            target_repo=context.ticket.parameters.get('target_repo', ''),
            branch_name=branch_name,
            title=f'Code Forge Pipeline {context.pipeline_id[:8]}',
            description=f'''Quality Score: {quality_score:.2f}

Artifacts: {len(context.generated_artifacts)}
Generation Method: {context.generation_method or 'TEMPLATE'}
Template: {context.selected_template.template_id if context.selected_template else 'N/A'}

---
Auto-generated by Neural Code Forge
Pipeline ID: {context.pipeline_id}'''
        )

        context.metadata['approval'] = 'manual'
        context.metadata['git_mr_url'] = mr_url

async def _commit_and_push(self, context: PipelineContext, auto_approved: bool) -> str:
    """
    Commita código gerado no repositório Git.

    Returns:
        Nome da branch criada
    """
    artifact = context.get_latest_artifact()
    if not artifact:
        logger.warning('no_artifact_to_commit')
        return None

    # Recuperar código do MongoDB
    from ..clients.mongodb_client import MongoDBClient
    mongodb_client = MongoDBClient(
        url=self.git_client.local_path.parent / 'mongodb_config',  # ou obter do contexto
        db_name='code_forge'
    )
    await mongodb_client.start()

    code_content = await mongodb_client.get_artifact_content(artifact.artifact_id)
    await mongodb_client.stop()

    # Determinar nome do arquivo
    language = artifact.language.lower()
    if language == 'python':
        filename = 'main.py'
    elif language in ('javascript', 'typescript'):
        filename = 'index.js' if language == 'javascript' else 'index.ts'
    elif language == 'go':
        filename = 'main.go'
    elif language == 'java':
        filename = 'Main.java'
    else:
        filename = f'main.{language}'

    # Criar branch
    branch_name = f'code-forge-{context.pipeline_id[:8]}'
    repo_url = context.ticket.parameters.get('target_repo', '')

    await self.git_client.create_branch(repo_url, branch_name)

    # Commitar artefatos
    artifacts = [
        {
            'path': filename,
            'content': code_content
        }
    ]

    # Adicionar README com metadados
    readme_content = f'''# Generated by Neural Code Forge

Pipeline ID: {context.pipeline_id}
Artifact ID: {artifact.artifact_id}
Generated at: {artifact.created_at.isoformat()}
Generation Method: {artifact.generation_method}
Confidence Score: {artifact.confidence_score}
'''
    artifacts.append({'path': 'README.md', 'content': readme_content})

    commit_sha = await self.git_client.commit_artifacts(
        repo_url=repo_url,
        branch=branch_name,
        artifacts=artifacts,
        message=f'Code Forge: {artifact.artifact_type} generated via {artifact.generation_method}'
    )

    # Push branch
    await self.git_client.push_branch(repo_url, branch_name)

    logger.info(
        'artifacts_committed_and_pushed',
        branch=branch_name,
        commit_sha=commit_sha[:8],
        files_count=len(artifacts)
    )

    return branch_name
```

```python
async def _cleanup_workspace(self, workspace: Path):
    """Limpa workspace após testes com retry."""
    try:
        import shutil
        max_retries = 3

        for attempt in range(max_retries):
            try:
                if workspace.exists():
                    shutil.rmtree(workspace)
                    logger.debug('workspace_cleaned', path=str(workspace), attempt=attempt+1)
                    return
            except PermissionError as e:
                if attempt < max_retries - 1:
                    logger.warning('workspace_cleanup_retrying', path=str(workspace), attempt=attempt+1)
                    await asyncio.sleep(1)
                else:
                    raise
            except Exception as e:
                logger.error('workspace_cleanup_failed', path=str(workspace), error=str(e))
                # Não falhar o pipeline se cleanup falhar
                break
    except Exception as e:
        logger.warning('workspace_cleanup_exception', error=str(e))
```

### 4. Validação de Licenças

**Novo arquivo:** `services/code-forge/src/services/license_validator.py`

```python
import structlog
from typing import List, Dict
from ..clients.mongodb_client import MongoDBClient
from ..models.artifact import ValidationResult, ValidationType, ValidationStatus

logger = structlog.get_logger()

# Licenças permitidas e problematicas
ALLOWED_LICENSES = {
    'MIT', 'Apache-2.0', 'BSD-2-Clause', 'BSD-3-Clause',
    'ISC', 'Python-2.0', 'LGPL-2.1', 'LGPL-3.0'
}

PROBLEMATIC_LICENSES = {
    'GPL-2.0': 'Copyleft - require code disclosure',
    'GPL-3.0': 'Copyleft - require code disclosure',
    'AGPL-3.0': 'Copyleft - network use triggers disclosure',
    'SSPL': 'Non-OSI approved - source-available only'
}

class LicenseValidator:
    """Validação de licenças de dependências"""

    def __init__(self, mongodb_client: MongoDBClient):
        self.mongodb_client = mongodb_client

    async def validate_licenses(
        self,
        artifact_id: str,
        language: str
    ) -> ValidationResult:
        """
        Valida licenças de dependências baseado no SBOM.

        Returns:
            ValidationResult com issues de licença encontrados
        """
        import time
        from datetime import datetime

        start_time = time.monotonic()

        # Buscar SBOM do MongoDB
        sbom = await self.mongodb_client.get_artifact_sbom(artifact_id)

        if not sbom:
            return ValidationResult(
                validation_type=ValidationType.COMPLIANCE_CHECK,
                tool_name='license-validator',
                tool_version='1.0.0',
                status=ValidationStatus.SKIPPED,
                score=1.0,
                issues_count=0,
                critical_issues=0,
                high_issues=0,
                medium_issues=0,
                low_issues=0,
                executed_at=datetime.now(),
                duration_ms=0
            )

        # Analisar licenças
        critical_issues = 0
        high_issues = 0
        problematic_deps = []

        for component in sbom.get('components', []):
            licenses = component.get('licenses', [])
            if not licenses:
                continue

            for license_entry in licenses:
                license_id = license_entry.get('id', {}).get('name', 'UNKNOWN')

                if license_id in PROBLEMATIC_LICENSES:
                    critical_issues += 1
                    problematic_deps.append({
                        'component': component.get('name', 'unknown'),
                        'version': component.get('version', 'unknown'),
                        'license': license_id,
                        'reason': PROBLEMATIC_LICENSES[license_id]
                    })

        duration_ms = int((time.monotonic() - start_time) * 1000)

        return ValidationResult(
            validation_type=ValidationType.COMPLIANCE_CHECK,
            tool_name='license-validator',
            tool_version='1.0.0',
            status=ValidationStatus.PASSED if critical_issues == 0 else ValidationStatus.FAILED,
            score=1.0 if critical_issues == 0 else 0.3,
            issues_count=len(problematic_deps),
            critical_issues=critical_issues,
            high_issues=high_issues,
            medium_issues=0,
            low_issues=0,
            executed_at=datetime.now(),
            duration_ms=duration_ms,
            report_uri=f'mongodb://sboms/{artifact_id}' if sbom else None
        )
```

```python
async def _run_nodejs_tests(self, workspace: Path, ticket_id: str) -> ValidationResult:
    """
    Executa testes para JavaScript/TypeScript usando npm test.

    Returns:
        ValidationResult com métricas dos testes
    """
    start_time = datetime.now()

    try:
        # Verificar se package.json existe
        package_json = workspace / 'package.json'
        if not package_json.exists():
            # Criar package.json mínimo
            package_json.write_text('''{
  "name": "generated-service",
  "version": "1.0.0",
  "scripts": {
    "test": "jest --coverage"
  },
  "devDependencies": {
    "jest": "^29.0.0"
  }
}
''')

        # Instalar dependências (npm ci --ignore-scripts)
        install_proc = await asyncio.create_subprocess_exec(
            'npm', 'ci', '--ignore-scripts',
            cwd=str(workspace),
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE
        )
        await install_proc.wait()

        # Executar testes
        cmd = ['npm', 'test', '--', '--coverage', '--json']
        proc = await asyncio.create_subprocess_exec(
            *cmd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
            cwd=str(workspace)
        )

        stdout, stderr = await asyncio.wait_for(
            proc.communicate(),
            timeout=self.test_timeout + 10
        )

        duration_ms = int((datetime.now() - start_time).total_seconds() * 1000)

        # Parse resultados Jest
        try:
            coverage_file = workspace / 'coverage' / 'coverage-final.json'
            if coverage_file.exists():
                import json
                coverage_data = json.loads(coverage_file.read_text())
                total_lines = coverage_data.get('total', {}).get('lines', {}).get('pct', 0)
                coverage_pct = total_lines / 100.0
            else:
                coverage_pct = 0.0
        except:
            coverage_pct = 0.0

        return ValidationResult(
            validation_type=ValidationType.UNIT_TEST,
            tool_name='jest',
            tool_version='29.0.0',
            status=ValidationStatus.PASSED if proc.returncode == 0 else ValidationStatus.FAILED,
            score=coverage_pct,
            issues_count=0 if proc.returncode == 0 else 1,
            critical_issues=1 if coverage_pct < self.min_coverage else 0,
            high_issues=0,
            medium_issues=0,
            low_issues=0,
            executed_at=start_time,
            duration_ms=duration_ms
        )

    except Exception as e:
        duration_ms = int((datetime.now() - start_time).total_seconds() * 1000)
        logger.error('nodejs_test_failed', error=str(e))
        return ValidationResult(
            validation_type=ValidationType.UNIT_TEST,
            tool_name='jest',
            tool_version='29.0.0',
            status=ValidationStatus.FAILED,
            score=0.0,
            issues_count=1,
            critical_issues=1,
            high_issues=0,
            medium_issues=0,
            low_issues=0,
            executed_at=start_time,
            duration_ms=duration_ms
        )
```

```python
# Em git_client.py

def list_tags(self) -> List[str]:
    """Lista todas as tags do repositório."""
    if not self.repo:
        return []
    return [tag.name for tag in self.repo.tags]

def checkout_tag(self, tag_name: str):
    """Faz checkout de uma tag específica."""
    if not self.repo:
        raise RuntimeError('Repositório não foi clonado')

    self.repo.git.checkout(tag_name)
    logger.info('tag_checked_out', tag=tag_name)
```
